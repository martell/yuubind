[−][src]Macro futures_test::assert_stream_done

macro_rules! assert_stream_done {
    ($stream:expr) => { ... };
}

Assert that the next poll to the provided stream will return an empty Poll::Ready signalling the completion of the stream.

Examples

use futures::stream;
use futures_test::future::FutureTestExt;
use futures_test::{
    assert_stream_pending, assert_stream_next, assert_stream_done,
};
use futures::pin_mut;

let stream = stream::once((async { 5 }).pending_once());
pin_mut!(stream);

assert_stream_pending!(stream);
assert_stream_next!(stream, 5);
assert_stream_done!(stream);
